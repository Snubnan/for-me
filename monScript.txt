  };

  function getTextNodeFromPoint(x, y) {
    let range = null;
    if (document.caretRangeFromPoint) range = document.caretRangeFromPoint(x, y);
    else if (document.caretPositionFromPoint) {
      const pos = document.caretPositionFromPoint(x, y);
      if (pos) { range = document.createRange(); range.setStart(pos.offsetNode, pos.offset); range.collapse(true); }
    }
    if (!range) return null;
    let node = range.startContainer;
    if (node && node.nodeType === Node.TEXT_NODE) return (node.nodeValue||'').trim().length > 0 ? node : null;
    if (node && node.nodeType === Node.ELEMENT_NODE) {
      const walker = document.createTreeWalker(node, NodeFilter.SHOW_TEXT, { acceptNode: n => (n.nodeValue||'').trim().length > 0 ? NodeFilter.FILTER_ACCEPT : NodeFilter.FILTER_REJECT });
      return walker.nextNode();
    }
    return null;
  }

            else ce.textContent = count.toString();
          }
          removeEditor(); state.editNode = null; state.editOriginal = null; return;
        }

    );

    const searchIcon = document.createElement('span');

      flex:1;background:rgba(255,255,255,0.07);border:1px solid rgba(255,255,255,0.12);

    body.appendChild(sep2);

      () => state.hideNonAdEnabled,

        state.hideNonAdEnabled = !state.hideNonAdEnabled;
        if (state.hideNonAdEnabled) { startHideNonAdObserver(); }
        else { stopHideNonAdObserver(); }
        toast(`Сховати не-#ad: ${state.hideNonAdEnabled ? 'ON 📣' : 'OFF'}`);
      }
    );
    body.appendChild(hideAdRow.row); rows.push(hideAdRow);

    // Separator
    const sep3 = document.createElement('div');
    sep3.style.cssText = 'height:1px;background:rgba(255,255,255,0.06);margin:5px 14px;';
    body.appendChild(sep3);

    // ---- Reset button ----
    const resetBtn = document.createElement('button');
    resetBtn.textContent = '↩ Скинути всі зміни';
    resetBtn.style.cssText = `
      display:block;width:calc(100% - 28px);margin:6px 14px 4px;
      padding:8px;border-radius:10px;
      border:1px solid rgba(255,59,48,0.30);
      background:rgba(255,59,48,0.10);
      color:#FF3B30;font-size:12.5px;font-weight:500;
      cursor:pointer;transition:all 0.15s;
      font-family:-apple-system,system-ui,sans-serif;
    `;
    resetBtn.onmouseenter = () => { resetBtn.style.background = 'rgba(255,59,48,0.20)'; resetBtn.style.borderColor = 'rgba(255,59,48,0.55)'; };
    resetBtn.onmouseleave = () => { resetBtn.style.background = 'rgba(255,59,48,0.10)'; resetBtn.style.borderColor = 'rgba(255,59,48,0.30)'; };
    resetBtn.onclick = resetAllChanges;
    body.appendChild(resetBtn);

    // ---- Hotkeys hint ----
    const hint = document.createElement('div');
    hint.style.cssText = 'padding:2px 14px 8px;';
    hint.innerHTML = `<span style="color:rgba(235,235,245,0.22);font-size:10px;line-height:1.5;">Ctrl+Alt+⌫ = скинути&nbsp;&nbsp;•&nbsp;&nbsp;повторний запуск = показати панель</span>`;
    body.appendChild(hint);

    const updatePanel = () => rows.forEach(r => r.refresh && r.refresh());

    document.documentElement.appendChild(panel);

    // Make draggable
    let dragging = false, dragOffX = 0, dragOffY = 0;
    header.style.cursor = 'grab';
    header.addEventListener('mousedown', (e) => {
      if (e.target === closeBtn) return;
      dragging = true;
      const rect = panel.getBoundingClientRect();
      dragOffX = e.clientX - rect.left;
      dragOffY = e.clientY - rect.top;
      header.style.cursor = 'grabbing';
      e.preventDefault();
    });
    document.addEventListener('mousemove', (e) => {
      if (!dragging) return;
      panel.style.left = `${Math.max(0, Math.min(window.innerWidth - panel.offsetWidth, e.clientX - dragOffX))}px`;
      panel.style.top  = `${Math.max(0, Math.min(window.innerHeight - panel.offsetHeight, e.clientY - dragOffY))}px`;
    });
    document.addEventListener('mouseup', () => { dragging = false; header.style.cursor = 'grab'; });
  };

  // ===== EVENT HANDLERS =====
  const onKeyDown = (e) => {
    if (e.ctrlKey && e.altKey && e.key === 'Backspace') {
      e.preventDefault(); e.stopPropagation();
      resetAllChanges(); return;
    }
    if (!state.ctrlEditorEnabled && !state.altModeEnabled) return;

    if (e.key === 'Control') { state.ctrlDown = true; ensureHiBox(); }
    if (e.key === 'Alt' && !altState.altDown && state.altModeEnabled) {
      altState.altDown = true; ensureAltHiBox();
      const el = document.elementFromPoint(state.mouseX, state.mouseY);
      const target = el && el.closest(ALT_TARGET_SELECTOR);
      altState.hoveredTarget = target || null;
      if (target) placeAltHiBox(target); else hideAltHiBox();
    }
  };

  const onKeyUp = (e) => {
    if (e.key === 'Control') { state.ctrlDown = false; hideHiBox(); }
    if (e.key === 'Alt') { altState.altDown = false; hideAltHiBox(); altState.hoveredTarget = null; }
  };

  const onMouseMove = (e) => {
    state.mouseX = e.clientX; state.mouseY = e.clientY;
    if (state.ctrlEditorEnabled && state.ctrlDown && !state.editor) {
      ensureHiBox();
      const el = document.elementFromPoint(e.clientX, e.clientY);
      placeHiBox(pickEditableTarget(el) || el);
    }
    if (state.altModeEnabled && altState.altDown) {
      const el = document.elementFromPoint(e.clientX, e.clientY);
      const target = el && el.closest(ALT_TARGET_SELECTOR);
      altState.hoveredTarget = target || null;
      if (target) placeAltHiBox(target); else hideAltHiBox();
    }
  };

  const onMouseDown = (e) => {
    if (state.altModeEnabled && altState.altDown && altState.hoveredTarget) {
      e.preventDefault(); e.stopPropagation();
      const el = altState.hoveredTarget;
      if (!altState.originalContents.has(el)) altState.originalContents.set(el, el.innerHTML);
      el.innerHTML = EMPTY_STATE_HTML;
      hideAltHiBox(); altState.hoveredTarget = null;
      toast('🚫 Порожній стан застосовано');
      return;
    }
    if (!state.ctrlEditorEnabled || !state.ctrlDown || state.editor) return;
    const el = document.elementFromPoint(e.clientX, e.clientY);
    const targetEl = pickEditableTarget(el);
    if (!targetEl) return;
    const tn = getTextNodeFromPoint(e.clientX, e.clientY);
    if (!tn) return;
    e.preventDefault(); e.stopPropagation();
    state.editNode = tn;
    state.editOriginal = tn.nodeValue ?? '';
    if (!state.edited.has(tn)) state.edited.set(tn, state.editOriginal);
    const ps = tn.parentElement;
    if (ps && (ps.tagName||'').toLowerCase() === 'span' && !state.spanClassBackup.has(ps)) state.spanClassBackup.set(ps, ps.className || '');
    makeEditor(e.clientX + 14, e.clientY + 14, state.editOriginal);
  };

  window.addEventListener('keydown', onKeyDown, true);
  window.addEventListener('keyup', onKeyUp, true);
  window.addEventListener('mousemove', onMouseMove, true);
  window.addEventListener('mousedown', onMouseDown, true);

  const uninstall = () => {
    try { window.removeEventListener('keydown', onKeyDown, true); } catch(e){}
    try { window.removeEventListener('keyup', onKeyUp, true); } catch(e){}
    try { window.removeEventListener('mousemove', onMouseMove, true); } catch(e){}
    try { window.removeEventListener('mousedown', onMouseDown, true); } catch(e){}
    cancelEdit(); stopSearchPoll(); stopScrollEmu(); removePhantom(); stopHideNonAdObserver();
    if (state.hiBox) { state.hiBox.remove(); state.hiBox = null; }
    if (altState.altHiBox) { altState.altHiBox.remove(); altState.altHiBox = null; }
    const panel = document.getElementById('__om_panel__');
    if (panel) panel.remove();
    window.__omTools = null;
    return { ok: true };
  };

  window.__omTools = { version: VERSION, uninstall };

  createPanel();
  toast('⚙ OM Tools v22 встановлено');
  return { ok: true, msg: 'installed', version: VERSION };
})();
